CREATE TABLE TBL_LOC_AREA (AREA_ID serial,AREA_TYPE_ID INTEGER NOT NULL ,CODE VARCHAR(30) NOT NULL ,NAME VARCHAR(200) NOT NULL ,CONSTRAINT TBL_LOC_AREA_PK PRIMARY KEY (AREA_ID),CONSTRAINT TBL_LOC_AREA_UK UNIQUE (CODE));
CREATE TABLE TBL_LOC_ADDRESS (ADDRESS_ID serial,COUNTRY_ID INTEGER NOT NULL ,STREET_NAME VARCHAR(250)  ,STREET_NR_FIRST VARCHAR(20)  ,POSTAL_CODE VARCHAR(30)  ,ADDRESS_LINE_1 VARCHAR(100)  ,ADDITIONAL_DATA_1 VARCHAR(100)  ,CONSTRAINT TBL_LOC_ADDRESS_PK PRIMARY KEY (ADDRESS_ID));
