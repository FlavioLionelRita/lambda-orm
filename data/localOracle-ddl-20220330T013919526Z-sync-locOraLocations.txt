CREATE TABLE TBL_LOC_AREA (AREA_ID NUMBER(10) NOT NULL,AREA_TYPE_ID NUMBER(10) NOT NULL ,CODE VARCHAR2(30) NOT NULL ,NAME VARCHAR2(200) NOT NULL ,CONSTRAINT TBL_LOC_AREA_PK PRIMARY KEY (AREA_ID),CONSTRAINT TBL_LOC_AREA_UK UNIQUE (CODE));
CREATE TABLE TBL_LOC_ADDRESS (ADDRESS_ID NUMBER(10) NOT NULL,COUNTRY_ID NUMBER(10) NOT NULL ,STREET_NAME VARCHAR2(250)  ,STREET_NR_FIRST VARCHAR2(20)  ,POSTAL_CODE VARCHAR2(30)  ,ADDRESS_LINE_1 VARCHAR2(100)  ,ADDITIONAL_DATA_1 VARCHAR2(100)  ,CONSTRAINT TBL_LOC_ADDRESS_PK PRIMARY KEY (ADDRESS_ID));
