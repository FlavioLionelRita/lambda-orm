CREATE TABLE Coordinates (id serial,geometryId INTEGER NOT NULL ,lat VARCHAR(80) NOT NULL ,long VARCHAR(80) NOT NULL ,alt VARCHAR(80)  ,CONSTRAINT Coordinates_PK PRIMARY KEY (id));
CREATE TABLE Geometries (id serial,shapeId INTEGER NOT NULL ,type VARCHAR(80) NOT NULL ,CONSTRAINT Geometries_PK PRIMARY KEY (id));
CREATE TABLE Properties (id serial,shapeId INTEGER NOT NULL ,MAPBLKLOT VARCHAR(80)  ,BLKLOT VARCHAR(80)  ,BLOCK_NUM VARCHAR(80)  ,LOT_NUM VARCHAR(80)  ,FROM_ST VARCHAR(80)  ,TO_ST VARCHAR(80)  ,STREET VARCHAR(80)  ,ST_TYPE VARCHAR(80)  ,ODD_EVEN VARCHAR(80)  ,CONSTRAINT Properties_PK PRIMARY KEY (id));
CREATE TABLE Shapes (id serial,type VARCHAR(80) NOT NULL ,CONSTRAINT Shapes_PK PRIMARY KEY (id));
ALTER TABLE Coordinates ADD CONSTRAINT Coordinates_geometry_FK FOREIGN KEY (geometryId) REFERENCES Geometries (id);
ALTER TABLE Geometries ADD CONSTRAINT Geometries_shape_FK FOREIGN KEY (shapeId) REFERENCES Shapes (id);
ALTER TABLE Properties ADD CONSTRAINT Properties_shape_FK FOREIGN KEY (shapeId) REFERENCES Shapes (id);
